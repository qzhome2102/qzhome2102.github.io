<head>
    <meta charset="UTF-8">
    <title>模拟百度蜘蛛获取网页源代码</title>
</head>
<body>
<?php
$url = "https://www.baidu.com"; // 要爬取的网页URL
$ua = "Mozilla/5.0 (+http://www.baidu.com/search/spider.htm)"; // 设置User-Agent为百度蜘蛛的User-Agent字符串

// 创建一个curl句柄
$ch = curl_init($url);

// 设置curl选项
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true); // 将curl_exec()获取的数据以字符串形式返回
curl_setopt($ch, CURLOPT_USERAGENT, $ua); // 设置User-Agent头字段
curl_setopt($ch, CURLOPT_FOLLOWLOCATION, true); // 开启自动重定向跟踪

// 执行curl请求
$html = curl_exec($ch);

// 关闭curl句柄
curl_close($ch);

// 输出获取到的HTML源代码
echo $html;
?>
</body>
</html>
